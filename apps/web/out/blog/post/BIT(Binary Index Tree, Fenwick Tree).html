	(1, 3, 6),
	(2, 2, 5),
	(1, 5, 2),
	(2, 3, 5),
]

def prefix_sum(i):
	result = 0
	while i > 0:
		result += tree[i]
		i -= (i & -i)
		return result

def interval_sum(s, e):
	return prefix_sum(e) - prefix_sum(s - 1)

def update(i, val):
	while i <= n:
		tree[i] += val
		i += (i & -i)
